    function MarkerOverlay() {
        var Overlay = function (map, text, latLng, ready) {
            this.map = map;
            this.text = text;
            this.latLng = latLng;
            this.ready = ready;
            this.div = null;
            this.setMap(this.map);
        }

        Overlay.prototype = new google.maps.OverlayView();

        $.extend(Overlay.prototype, {
            onAdd: function () {
                var div = $('<div/>').html(this.text).css('position', 'absolute');
                $(this.getPanes().overlayImage).append(div);
                this.div = div;
                if (typeof this.ready === 'function') {
                    this.ready.call(this);
                }
            },
            draw: function () {
                var latLng = this.getProjection().fromLatLngToDivPixel(this.latLng);
                this.div.css({
                    'left': latLng.x - this.div.width() / 2,
                    'bottom': -latLng.y
                });
            },
            onRemove: function () {
                this.div.remove();
                this.div = null;
            }
        });

        return Overlay;
    }

                        <li><b>坐车</b> 东京的站名几乎都是汉字，对中国人很方便，坐车时注意看站名就行，但是光靠听语音报站可能完全不知道是哪里。</li>      
                    
                        <li><b>消费税</b> 在东京无论您消费了什么，日本政府都要征收您支付额5%的消费税，所以不要惊讶。</li>      
                    
                        <li><b>海关</b> 无论何种通货货币都可以携带进出东京，但是如果携带的是100万日元以上的，或者是与其等值的支票和有价证券时，都必须向海关申报。</li>      
                    
                        <li><b>酒店</b> 东京的酒店小而精致，但是按人头算的，一个人入住和2个人入住，价格不一样。</li>      
                    
                        <li><b>就餐</b> 在日本，吃鱼时不要将鱼刺直接放在桌子上，请放在盘子里。</li>      
                    
                        <li><b>禁烟</b> 东京几乎全市都严格禁烟，在街上也不能随便抽烟，很多地方都在路面上贴有中英日韩四国文字的禁烟标识。</li>      
                    
                        <li><b>禁忌</b> 日本人忌讳“4”和“9”。因为“4”的读音和“死”相同；“9”和“苦”的读音相同，“9”意味着苦。</li>      
                    
                        <li><b>生活习惯</b> 日本人对守时的要求很严格，人们都认为不该因为迟到而浪费别人的时间，如果因为某种原因不可避免地迟到时，必须要打电话告诉对方。</li>      
                    
                        <li><b>存放行李</b> 东京的车站、机场和部分百货商店设有投币储物柜，可存放各种行李，对自助游客非常方便。一般需要300-500日元/天。</li>      
                    
                        <li><b>公共交通</b> 无论是在室内外或者是地铁等交通工具上，都不要发出太大声响，尤其讲电话要小声，日本人习惯于乘地铁时不接听手机。</li>      
                    
                        <li><b>安全</b> 女性应该小心，在东京拥挤的列车中会有色狼出没，但他们通常瞄准的是当地女性。</li>      
                    
                        <li><b>夜店</b> 六本木和新宿歌舞伎町的酒吧和夜店招揽顾客的人可能相当咄咄逼人。最好不要跟随他们，有过往饮料里下药然后偷窃甚至殴打的先例。</li>      
                    
                        <li><b>脱鞋</b> 在东京，进入榻榻米、私人居所或寺庙大厅的时候，要记得脱掉自己的鞋子，以保持屋内的地板不被街上带来的灰尘弄脏。</li>

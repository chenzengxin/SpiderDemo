    function MarkerOverlay() {
        var Overlay = function (map, text, latLng, ready) {
            this.map = map;
            this.text = text;
            this.latLng = latLng;
            this.ready = ready;
            this.div = null;
            this.setMap(this.map);
        }

        Overlay.prototype = new google.maps.OverlayView();

        $.extend(Overlay.prototype, {
            onAdd: function () {
                var div = $('<div/>').html(this.text).css('position', 'absolute');
                $(this.getPanes().overlayImage).append(div);
                this.div = div;
                if (typeof this.ready === 'function') {

                    'left': latLng.x - this.div.width() / 2,
                    'bottom': -latLng.y
                });
            },
            onRemove: function () {
                this.div.remove();
                this.div = null;
            }
        });

    }

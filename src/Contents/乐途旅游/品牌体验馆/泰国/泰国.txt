    function MarkerOverlay() {
        var Overlay = function (map, text, latLng, ready) {
            this.map = map;
            this.text = text;
            this.latLng = latLng;
            this.ready = ready;
            this.div = null;
            this.setMap(this.map);
        }

        Overlay.prototype = new google.maps.OverlayView();

        $.extend(Overlay.prototype, {
            onAdd: function () {
                var div = $('<div/>').html(this.text).css('position', 'absolute');
                $(this.getPanes().overlayImage).append(div);
                this.div = div;
                if (typeof this.ready === 'function') {
                    this.ready.call(this);
                }
            },
            draw: function () {
                var latLng = this.getProjection().fromLatLngToDivPixel(this.latLng);
                this.div.css({
                    'left': latLng.x - this.div.width() / 2,
                    'bottom': -latLng.y
                });
            },
            onRemove: function () {
                this.div.remove();
                this.div = null;
            }
        });

        return Overlay;
    }

                        <li><b>还价</b> 百货公司的商品通常不二价，议价幅度很小，但在一般的商店则可以讨价还价，有九折到六折的弹性；一旦店家答应了你的还价要求，请不要食言。</li>      
                    
                        <li><b>禁赌</b> 泰国是完全禁赌国家，即使在旅馆房间也不要玩牌或打麻将，若被酒店保安发觉，报案后有口难辩，甚至有被敲诈的危险。</li>      
                    
                        <li><b>洗漱用品</b> 泰国及普吉岛等东南亚国家普遍没有准备盥洗用具如：牙膏、牙刷等，需自备。</li>      
                    
                        <li><b> 晚上出游小心</b> 晚上出游，对不认识的人，不要随便轻信，招致不必要的损失。 </li>      
                    
                        <li><b>搭乘快艇</b> 搭乘快艇时请依序上下，听从工作人员指挥。</li>      
                    
                        <li><b>乘车安全</b> 车辆行进时勿站立、行走或更换座位，头、手勿伸出窗外，上下车时注意来车方向，以免发生危险。 </li>      
                    
                        <li><b>水上游乐活动</b> 活动若具有刺激性，身体状况不佳者请勿参加。不要贪图便宜去玩一些没有保障的水上游乐活动，当心被敲诈。 </li>      
                    
                        <li>海边戏水请勿超越安全警戒线。 </li>      
                    
                        <li><b>搭乘须知</b> 搭乘船只请务必穿着救生衣。 乘坐快艇出海时，年龄稍大的游客不要坐在船头部分，以免快艇运行时的颠簸，伤到脊椎。</li>      
                    
                        <li><b>游泳池</b> 游泳池未开放时请勿擅自入池游泳，并切记勿单独入池。 </li>      
                    
                        <li><b>忌带榴莲</b> 榴莲属于有气味的水果，不可带到游览车上或住宿的旅馆酒店内。</li>      
                    
                        <li><b>勿露财</b> 切勿在公共场合露财，购物时也勿当众清数钞票。 </li>      
                    
                        <li><b>携带须知</b> 请妥善保管护照、证件及个人贵重物品，请勿放于巴士、旅馆、房间内以及大行李中，并谨防扒手及陌生人搭讪。 </li>      
                    
                        <li><b>贵重物品</b> 贵重物品请托放至饭店保险箱，如需随身携带切勿离手，例如享用自助餐时，若将钱包放在餐桌上，很容易让扒手得逞。 </li>      
                    
                        <li><b>住店须知</b> 住饭店时随时将房门扣上安全锁，以策安全；勿在灯上晾衣物；勿在床上吸烟，听到警报器响，请由紧急出口迅速离开。 </li>      
                    
                        <li><b>乘机须知</b> 搭乘飞机时，请随时扣紧安全带，以免乱流影响安全。 </li>

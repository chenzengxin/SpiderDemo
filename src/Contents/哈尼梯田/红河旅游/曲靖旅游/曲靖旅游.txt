    function MarkerOverlay() {
        var Overlay = function (map, text, latLng, ready) {
            this.map = map;
            this.text = text;
            this.latLng = latLng;
            this.ready = ready;
            this.div = null;
            this.setMap(this.map);
        }

        Overlay.prototype = new google.maps.OverlayView();

        $.extend(Overlay.prototype, {
            onAdd: function () {
                var div = $('<div/>').html(this.text).css('position', 'absolute');
                $(this.getPanes().overlayImage).append(div);
                this.div = div;
                if (typeof this.ready === 'function') {
                    this.ready.call(this);
                }
            },
            draw: function () {
                var latLng = this.getProjection().fromLatLngToDivPixel(this.latLng);
                this.div.css({
                    'left': latLng.x - this.div.width() / 2,
                    'bottom': -latLng.y
                });
            },
            onRemove: function () {
                this.div.remove();
                this.div = null;
            }
        });

    }

	            $.getJSON('http://my.lotour.com/login/login/islogin?callback=?&' + (new Date()), function (data) {
	                if (data.uid > 0) {
	                    showHelp(regionid, rname);
	                }
	                else {
	                    jQuery.lotour.user.showLoginV3(function (data) {
	                        SetTopState();
	                        showHelp(regionid, rname);
	                    }, function (data) {
	                        SetTopState();
	                        showHelp(regionid, rname);
	                    }, 1, location.href.toString().replace("#", "") + "?rid=" + regionid + "&ReturnAction=ask");
	                }
	            });
	        });
	    });
	    function showHelp(regionid, rname) {
	        $('#ask_topic').fadeIn(500).loadAskPanel(0);
	        $('.layerBg').css('height', $(document).height()).show();
	        $('#txtAskRegionSelect1').val(rname).attr('disabled', 'disabled');
	        $('#hidAskTopicRegionId1').val(regionid);
	        $.ajax({
	            cache: false,
	            type: "GET",
	            dataType: 'jsonp',
	            jsonp: 'callback',
	            url: "http://my.lotour.com/i/DPlace/AskTopic/GetBeenCount/",
	            data: { regionId: regionid },
	            success: function (data) {
	                data = data.Result.Data;
	                if (data.count > 0) {
	                    $("#BeenMsg").html(data.count + "专家为您解答");
	                }
	            },
	            error: function (xhr, textStatus, errorThrown) {
	            }
	        });
	    }

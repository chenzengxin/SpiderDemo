function isMobile(str) {
    if (/^13\d{9}$/g.test(str) || (/^15\d{9}$/g.test(str)) || (/^18\d{9}$/g.test(str)) || (/^17\d{9}$/g.test(str))) {
        return true;
    }
    else {
        return false;
    }
}

	function stripscript(s)
	{
	var pattern = new RegExp("[`~!@#$^&*()%=|{}':;',\\[\\].<>/?~！@#￥……&*（）——|{}【】‘；：”“'。，、？]")
	var rs = "";
	for (var i = 0; i < s.length; i++) {
	rs = rs+s.substr(i, 1).replace(pattern, '');
	}
	if(rs==s){
		return false;
	}
	return true;
	}

	}

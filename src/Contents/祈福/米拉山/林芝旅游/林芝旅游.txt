    function MarkerOverlay() {
        var Overlay = function (map, text, latLng, ready) {
            this.map = map;
            this.text = text;
            this.latLng = latLng;
            this.ready = ready;
            this.div = null;
            this.setMap(this.map);

        Overlay.prototype = new google.maps.OverlayView();

        $.extend(Overlay.prototype, {
            onAdd: function () {
                var div = $('<div/>').html(this.text).css('position', 'absolute');
                $(this.getPanes().overlayImage).append(div);
                this.div = div;
                if (typeof this.ready === 'function') {

                    'bottom': -latLng.y
                });
            },
            onRemove: function () {
                this.div.remove();
                this.div = null;
            }
        });

    }

                        <li><b>高原反应</b> 从高原回到平原时，为预防醉氧，最好要逐渐地下降高度。回到低海拔地区后，可以尽早安排休息，以重新适应当地环境。</li>      
                    
                        <li><b>必备零食</b> 尽可能吃高糖的食品，例如巧克力、士力架和糖果，牛奶和酥油茶也可提供足够的热量。葡萄粉和葡萄干能迅速转化为葡萄糖，能减轻高翻。</li>      
                    
                        <li><b>服用药物</b> 提前7-10天服用预防高原反应的药品，例如高原红景天、诺迪康胶囊、百服宁、丹参丸等。</li>      
                    
                        <li><b>谨慎洗澡</b> 虽然现在藏区洗澡比较方便，但建议容易着凉的人，刚进入藏区还是少洗，等过了高原反应后再洗。</li>      
                    
                        <li><b>通麦天险</b> 位于波密和林芝八一镇之间，是典型的泥石流区，道路凶险，山体滑坡，路面下陷，飞石砸落无处不在，路况一直不好千万小心。</li>      
                    
                        <li><b>塌方泥石流</b> 5-9月雨季，也要时刻警惕路上塌方和泥石流。</li>      
                    
                        <li><b>进入墨脱</b> 进墨脱要准备足够现金和一些食物，10天约1500-2000元，交通不便沿途的食品都很贵，住宿简陋但费用较低。</li>      
                    
                        <li><b>墨脱防蚂蟥</b> 5月中旬后墨脱可勉强进入，回程去波密要翻雪山和徒步，最好避开7-9月的雨季，因为蚂蝗很多，个性墨脱徒步部分有详细介绍。</li>      
                    
                        <li><b>火车进藏</b> 会被要求填写《旅客健康登记卡》，表明自己对于进藏后有可能遇到的高原反应等身体状况充分知晓，并自行承担责任。</li>      
                    
                        <li><b>林芝需要办理边防证的地方</b> 米林大部分地区；郎县金东乡；察隅县上下察隅、竹瓦根、察瓦龙；墨脱县墨脱、背崩、德兴、格当、达木洛巴。</li>      
                    
                        <li><b>边防证办理</b> 可在户口所在地公安局办理，或者去拉萨办理。林芝办理边防证相对麻烦，可以交由当地旅行社办理。</li>      
                    
                        <li><b>拍照注意</b> 近距离拍摄人，尤其是僧侣、妇女，一定要事先征得对方的同意，许多寺庙内部是不允许拍摄的。在一些景点会有当地藏民请求付钱后再拍照。</li>      
                    
                        <li><b>作客习俗</b> 在藏民家做客，切记不要乱摸小孩的头。主人请你喝茶或饮酒时，如果确实不能喝，要婉言谢绝。</li>      
                    
                        <li><b>藏历新年</b> 忌讳说没有、穷、病、死、杀、鬼等不吉利的话。忌讳打碎碗、杯、盘、碟等用具。忌吃有馅食物，并且藏族人不吃狗肉、猴肉、马肉和驴肉。</li>      
                    
                    </ul>

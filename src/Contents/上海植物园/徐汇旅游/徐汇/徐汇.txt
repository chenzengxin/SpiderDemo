    function MarkerOverlay() {
        var Overlay = function (map, text, latLng, ready) {
            this.map = map;
            this.text = text;
            this.latLng = latLng;
            this.ready = ready;
            this.div = null;

        Overlay.prototype = new google.maps.OverlayView();

                    'bottom': -latLng.y
                });
            },
            onRemove: function () {
                this.div.remove();
                this.div = null;
